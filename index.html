// Bot runs every 5 minutes. Open-source. Fully auditable.
// Below is the representative pseudo-code of the protocol's
// keeper loop. Production code is published on GitHub at launch.

const TREASURY  = 'GRefsnCKo9QELRL7L2bQnkKM6u1Y16VgmkX8U4HDAcFT';
const QQQX_MINT = 'Xs8S1uUs1zvS2p7iwtsG3b6fkhpvmwz4GYU3gWAmWHZ';
const PUMPFUN   = '6EF8rrecthR5Dkzon8Nwu78hRvfCKubJ14M5uBEwF6P';
const MIN_SOL   = 0.05;   // skip cycle if too little SOL (avoid slippage)

async function runCycle() {

  // 1. Scan recent txs landing in the treasury wallet
  const txs = await rpc.getSignaturesForAddress(TREASURY);

  // 2. Sum SOL inflows that came from pump.fun creator-fee claims
  //    (manual claims by the operator land here the same way)
  let feeSol = 0;
  for (const tx of txs) {
    if (calledBy(tx, PUMPFUN)) {
      feeSol += extractSolTransfer(tx, TREASURY);
    }
  }

  // 3. Skip the cycle if accumulated SOL is too small
  if (feeSol < MIN_SOL) return;

  // 4. Swap SOL → QQQx via Jupiter's best route
  const route = await jupiter.getRoute({
    inputMint:  SOL_MINT,
    outputMint: QQQX_MINT,
    amount:     feeSol * LAMPORTS_PER_SOL,
  });
  const { outAmount } = await jupiter.swap(route);

  // 5. Snapshot $SPCXBANK holders + balances at this block
  const holders = await helius.getTokenHolders(SPCXBANK_MINT);
  const supply  = holders.reduce((s, h) => s + h.balance, 0);

  // 6. Compute pro-rata share + batch-transfer QQQx to each wallet
  const drops = holders.map(h => ({
    to:     h.address,
    amount: (h.balance / supply) * outAmount,
  }));
  await batchTransfer(drops, QQQX_MINT);

  // 7. Done. Next cycle in 5 min.
}

setInterval(runCycle, 5 * 60 * 1000);
